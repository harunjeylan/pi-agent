├── science/         # Science topics

├── permanent/       # Permanent notes (Zettelkasten)

├── fleeting/        # Quick thoughts

sources/             # Source materials
├── books/           # Book notes
├── articles/        # Article summaries
├── courses/         # Course notes
├── podcasts/        # Podcast notes
└── videos/          # Video notes
templates/           # Note templates
├── book-note/
├── article-note/
├── daily-note/

daily/               # Daily notes & journals
├── yyyy-mm/         # Monthly folders
│   └── yyyy-mm-dd.md
reviews/             # Periodic reviews
├── weekly/          # Weekly reviews
├── monthly/         # Monthly reviews
└── yearly/          # Yearly reviews
